def to_hash(keys_case = :underscore)
        hash = super(keys_case)
        hash[get_attribute_key('aggregation', keys_case)] = AGGREGATIONS[self.aggregation.to_s]
        hash[get_attribute_key('time', keys_case)] = time_to_hash(keys_case)
        return hash
end
